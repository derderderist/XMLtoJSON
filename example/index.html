List of examples for using XMLtoJSON

XML source

The xml example (xml/example.xml) file contains:

Default transformation

      var data = new XMLtoJSON({
        url: 'xml/example.xml'
      });

Generated JSON

Custom parameters

Include namespace information

        var data = new XMLtoJSON({
          url: 'xml/example.xml',
          namespaces: true
        });

Change value identifier

        var data = new XMLtoJSON({
          url: 'xml/example.xml',
          valueIdentifier: 'text'
        });

Change attribute identifier

        var data = new XMLtoJSON({
          url: 'xml/example.xml',
          attributeIdentifier: '@'
        });

Detect types (JSON is backparsed as String for output preview)

        var data = new XMLtoJSON({
          url: 'xml/example.xml',
          detectTypes: true
        });

Add a custom filter

        var data = new XMLtoJSON({
          url: 'xml/example.xml',
          filter: function(value){
            if(value === 'Thomas'){
              return 'Michael';
            }
            else{
              return value;
            }
          }
        });

Add a custom fallback method if XML is invalid or could not requested

        var data = new XMLtoJSON({
          url: 'xml/non-existing.xml',
          fallback: function(e){
            alert(e.code + ' - ' + e.message);
          }
        });

Set empty attribute (attribute="") and empty node value (<node/>) to null

        var data = new XMLtoJSON({
          url: 'xml/example.xml',
          emptyValuesAsNull : true
        });

Remove 'response' node and move all children one level up

        var data = new XMLtoJSON({
          url: 'xml/example.xml',
          modify: {
            'response.*' : '.'
          }
        });

Remove 'response.context' node and move all children into root

        var data = new XMLtoJSON({
          url: 'xml/example.xml',
          modify: {
            'response.context.*' : '.'
          }
        });

Remove 'response.context' node and move all children into root. Move 'response.version' as 'version' into root.

        var data = new XMLtoJSON({
          url: 'xml/example.xml',
          modify: {
            'response.context.*' : '.',
            'response.version' : 'version'
          }
        });

Remove 'response.context' node and move all children into root. Move 'response.version' as 'version' into root. Node 'response.context' and 'response' is now empty – clear node with param clearEmptyNodes.

        var data = new XMLtoJSON({
          url: 'xml/example.xml',
          modify: {
            'response.context.*' : '.',
            'response.version' : 'version'
          },
          clearEmptyNodes : true
        });

Rename 'response' node into 'info'.

        var data = new XMLtoJSON({
          url: 'xml/example.xml',
          modify: {
            'response' : 'info'
          }
        });

Move 'response.context.misc.software' into new non-existing path 'response.info.misc.tools' (rename 'software' to 'tools').

        var data = new XMLtoJSON({
          url: 'xml/example.xml',
          modify: {
            'response.context.misc.software' : 'response.info.misc.tools'
          },
          clearEmptyNodes : true
        });

Save all attributes with name 'key' of 'response.context.texts.text' into 'keys'.

        var data = new XMLtoJSON({
          url: 'xml/example.xml',
          modify: {
            'response.context.texts.text._key' : 'keys'
          }
        });

Save the first user as 'response.first_user'.

        var data = new XMLtoJSON({
          url: 'xml/example.xml',
          modify: {
            'response.context.users.user[0]' : 'response.first_user'
          }
        });

Methods

Add custom nodes.

        var data = new XMLtoJSON({
          url: 'xml/example.xml'
        });
        data.createNodes('response.context.info.countries.asia.india');

Remove node.

        var data = new XMLtoJSON({
          url: 'xml/example.xml'
        });
        data.remove('response.context');

Remove specified entry of node.

        var data = new XMLtoJSON({
          url: 'xml/example.xml'
        });
        data.remove('response.context.users.user[0]');

Result

Search and find in JSON

The data.find() method is a powerful way to access nodes (optionally with conditions). There is a logic implemented which allows a string notation without knowledge of the different node types – this means that you can access all Arrays inside of some other Arrays. It is not necessary to use a Array index. The following return statements are not parsed.

Get all 'some' nodes in each 'response.context.misc.something.some'.

        var data = new XMLtoJSON({
          url: 'xml/example.xml'
        });
        data = data.find('response.context.misc.something.some');

Get all 'something' nodes.

        var data = new XMLtoJSON({
          url: 'xml/example.xml'
        });
        data = data.find('response.context.misc.something');

Get all 'user' in 'response.context.users'.

        var data = new XMLtoJSON({
          url: 'xml/example.xml'
        });
        data = data.find('response.context.users.user');

Get names of all 'user'.

        var data = new XMLtoJSON({
          url: 'xml/example.xml'
        });
        data = data.find('response.context.users.user.name.$');

Get all 'user' which are admins.

        var data = new XMLtoJSON({
          url: 'xml/example.xml'
        });
        data = data.find('response.context.users.user', 'admin.$==true');

Get all 'texts' with a key attribute starting with 'general'.

        var data = new XMLtoJSON({
          url: 'xml/example.xml'
        });
        data = data.find('response.context.texts.text', '_key=~/^general/');

Get all 'texts' with a value of 'My company'.

        var data = new XMLtoJSON({
          url: 'xml/example.xml'
        });
        data = data.find('response.context.texts.text', '$==My Company');

Get all 'user' with a age of >25 years.

        var data = new XMLtoJSON({
          url: 'xml/example.xml'
        });
        data = data.find('response.context.users.user', 'age.$>=25');

Get all 'user' with are 30 years old.

        var data = new XMLtoJSON({
          url: 'xml/example.xml',
          detectTypes: true
        });
        data = data.find('response.context.users.user', 'age.$==30');

Get all 'something' with a 'some.other.url' with value www.google.com.

        var data = new XMLtoJSON({
          url: 'xml/example.xml',
          detectTypes: true
        });
        data = data.find('response.context.misc.something', 'some.other.url.$==www.google.com');

Get all 'software' which are not named 'Adobe Photoshop'.

        var data = new XMLtoJSON({
          url: 'xml/example.xml',
          detectTypes: true
        });
        data = data.find('response.context.misc.software.name', '$!=Adobe Photoshop');

Get all 'users' if first 'user' has name 'Thomas'.

        var data = new XMLtoJSON({
          url: 'xml/example.xml',
          detectTypes: true
        });
        data = data.find('response.context.users', 'user[0].name.$==Thomas');
